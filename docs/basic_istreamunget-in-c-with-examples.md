// C++ code for basic_istream::unget()
#include <bits/stdc++.h>
using namespace std;

// Driver code
int main()
{
    // Declare string stream
    istringstream gfg("GeeksforGeeks");

    char a = gfg.get();
    if (gfg.unget()) {
        char b = gfg.get();
        cout << "We got: " << a << endl;
        cout << "After ungetting the "
             << "character once again"
             << " we got: "
             << b << endl;
    }

    return 0;
}
